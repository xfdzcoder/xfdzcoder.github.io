废话不多说，上案例代码：

public class Main {

    public static void main(String[] args) {

        // 并且实现 F 接口
        // 该代理类的方法调用都将被转发到 匿名内部类，然后打印方法名称
        F f = (F) Proxy.newProxyInstance(
                Main.class.getClassLoader(),
                new Class[]{F.class},
                new InvocationHandler() {
                    @Override

                        System.out.println(method.getName());
                        return null;
                    }
                }

        f.m();
    }

}

    void m();
